To run

  # 1. Start TimescaleDB
  docker compose -f docker/docker-compose.yml up timescaledb -d

  # 2. Copy .env and set DATABASE_URL (already defaults correctly for Docker)
  copy .env.example .env

  # 3. Run migrations (from api/)
  cd api && uv run alembic upgrade head

  # 4. Start the API
  uv run fastapi dev src/pulse_api/main.py

  # 5. Seed
  curl -X POST http://localhost:8000/dev/seed | python3 -m json.tool
